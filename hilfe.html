Test von Funktionsaufrufen im Schleifenkopf

Aufruf von count() im Schleifenkopf

<?php
$array = [];
for($i = 0; $i < 1000000; $i++) {
    $array[] = $i;
}
for($i = 0; $i < count($array); $i++) {
    true;
}

Aufruf von count() vor dem Schleifenkopf

<?php
$array = [];
for($i = 0; $i < 1000000; $i++) {
    $array[] = $i;
}
$count = count($array);
for($i = 0; $i < $count; $i++) {
    true;
}

Return by Reference

<?php
class Config
{
    private $values = [];

    public function getValues() {
        return $this->values;
    }
}

$config = new Config();

$config->getValues()['test'] = 'test';
echo $config->getValues()['test'] . PHP_EOL;

// 1. Alternative eigene Variable

Autoloader

<?php

function my_autoloader ($class) {
    if (strpos($class, 'My_') === 0) {
        $split = strpos($class, '_');
        $file = substr($class, $split);
        include 'application/' . $file . '.php';
    }
}

spl_autoload_register('my_autoloader');

// alterativ auch mit lambda

spl_autoload_register( function ($class) {
    if (strpos($class, 'My_') === 0) {
        $split = strpos($class, '_');
        $file = substr($class, $split + 1);
        include 'application/' . $file . '.php';
    }
});

Magic Get- und Set-Methoden

<?php

class Klasse {
    private $_array = [];
    public function __get($key) {
        echo $key . PHP_EOL;
        if ($key == 'bla') {
            return 'blub';
        }
        return $this->_array[$key];
    }

    public function __set($key, $value) {
        $this->_array[$key] = strrev($value);
    }
}

$obj = new Klasse();

$obj->eins = 'eins';
$obj->bla = 'bla';
$obj->zwei = 'zwei';

echo $obj->eins . PHP_EOL;
echo $obj->bla . PHP_EOL;
echo $obj->zwei . PHP_EOL;

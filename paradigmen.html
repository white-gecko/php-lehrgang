Programmierparadigmen

Imperativ, Funktional, Objektorientiert

Imperative Programmierung

<?php
function p ($n) {
    $b = 1;
    while ($n > 0) {
        $b = $n * $b;
        $n = $n - 1;
    }
    return $b;
}
echo p(4) . PHP_EOL;

Funktionale Programmierung

<?php
function f ($n) {
    if ($n > 0) {
        return $n * f($n - 1);
    } else if ($n == 0) {
        return 1;
    }
}
echo f(4) . PHP_EOL;

Objektorientierte Programmierung

<?php
class Boat {
    public $length;
    private $_vMax;
    public function __construct ($vMax, $length) {
        $this->_vMax = $vMax;
        $this->length = $length;
    }
    public function isLongerThan ($other) {
        if ($this->length > $other->length) { return true; }
        else { return false; }
    }
}

$a = new Boat(10, 420);
$b = new Boat(5, 230);

if ($a->isLongerThan($b)) {
    echo 'A ist laenger';
} else {
    echo 'A ist nicht laenger';
}

echo PHP_EOL;
